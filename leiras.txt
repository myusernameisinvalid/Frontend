import { useState, useEffect } from 'react';
// 8. és 9. FELADAT: A useParams segít kiolvasni az URL-ből a játékos egyedi ID-ját (:id)
import { useParams, Link } from 'react-router-dom';
import axios from 'axios';

export default function Player() {
  const { id } = useParams(); // URL-ből kinyert azonosító variable
  const [player, setPlayer] = useState(null);

  useEffect(() => {
    // 9. FELADAT: Egyetlen könyv/játékos lekérdezése a /darts/{id} REST API végpontról
    axios.get(`https://localhost:5000/darts/${id}`)
      .then(response => {
        setPlayer(response.data);
      })
      .catch(error => {
        console.error("Hiba az egyedi adatlap betöltésekor:", error);
      });
  }, [id]);

  // Biztonsági ellenőrzés, amíg az aszinkron adat megérkezik a backendről
  if (!player) {
    return <div className="text-center mt-5"><div className="spinner-border text-primary"></div></div>;
  }

  return (
    <div className="container py-5">
      <div className="card mx-auto shadow-lg border-0" style={{ maxWidth: '700px' }}>
        <div className="row g-0 align-items-center">
          
          {/* 5. FELADAT: Játékos képének megjelenítése */}
          <div className="col-md-5 p-4 text-center">
            <img src={player.image_url} alt={player.nev} className="img-fluid rounded shadow-sm" />
          </div>
          
          {/* Adatok listázása */}
          <div className="col-md-7">
            <div className="card-body p-4">
              <h2 className="card-title fw-bold text-primary mb-3">{player.nev}</h2>
              <hr />
              <p className="card-text fs-5"><strong>Azonosító (ID):</strong> {player.id}</p>
              <p className="card-text fs-5"><strong>Születési dátum:</strong> {player.birth_date}</p>
              <p className="card-text fs-5"><strong>Világbajnoki címek száma:</strong> <span className="badge bg-success fs-6">{player.world_ch_won} db</span></p>
              <p className="card-text">
                <strong>Hivatalos profil:</strong> <br />
                <a href={player.profile_url} target="_blank" rel="noreferrer" className="text-break">{player.profile_url}</a>
              </p>
              
              {/* Akció gombok alsó elrendezése */}
              <div className="d-flex gap-3 mt-4">
                
                {/* 7. FELADAT: Vissza navigáláshoz ikon elhelyezése és funkció biztosítása a főoldalra */}
                <Link to="/" className="btn btn-secondary d-flex align-items-center gap-2">

                  <i className="bi bi-pencil-fill fs-5"></i> Módosítás
                </button>
              </div>

            </div>
          </div>
        </div>
      </div>
    </div>
  );

AddPlayer.jsx:
import { useState } from 'react';
// 11. FELADAT: A useNavigate hook szükséges az adatok sikeres elküldése utáni automatikus visszairányításhoz
import { useNavigate } from 'react-router-dom';
import axios from 'axios';

export default function AddPlayer() {
  const navigate = useNavigate();

  // 14. FELADAT MEZŐK: Állapotváltozók (states) létrehozása az űrlap mezőinek követéséhez
  const [id, setId] = useState('');
  const [nev, setNev] = useState('');
  const [birthDate, setBirthDate] = useState('');
  const [worldChWon, setWorldChWon] = useState('');
  const [profileUrl, setProfileUrl] = useState('');
  const [imageUrl, setImageUrl] = useState('');

  // Az űrlap elküldésekor lefutó logikai függvény
  const handleSubmit = (e) => {
    e.preventDefault(); // Megakadályozza az oldal alapértelmezett böngésző-újratöltését

    // A 14. feladatban kért struktúra szerint felépített küldendő adatobjektum
    const playerPayload = {
      id: Number(id), // Számmá alakítás az adatbázis formátum kedvéért
      nev: nev,
      birth_date: birthDate,
      world_ch_won: Number(worldChWon), // Számmá alakítás
      profile_url: profileUrl,
      image_url: imageUrl
    };

    // 13. FELADAT: Új játékos adatainak elküldése a kiszolgáló szerver /darts URL-jére POST metódussal
    axios.post('https://localhost:5000/darts', playerPayload)
      .then(response => {
        // 12. FELADAT (1. rész): A backend által visszaküldött válasz sikeres megjelenítése alert ablakban
        alert("Sikeres mentés! A backend válasza: Játékos rögzítve.");
        
        // 11. FELADAT: Az adatok sikeres átküldése után automatikusan navigáljon vissza a főoldalra
        navigate('/');
      })
      .catch(error => {
        // 12. FELADAT (2. rész): Hiba esetén lekezeli a hibát és strukturáltan kiírja a console-ba
        console.error("Hiba történt a POST küldés során:", error);
        alert("Hiba lépett fel a küldés közben, részletek a fejlesztői konzolban!");
      });
  };

  return (
    <div className="container my-4" style={{ maxWidth: '600px' }}>
      
      {/* 15. FELADAT: Behelyettesítve a vizsgázó neve helyére a saját neved, hogy látszódjon a tel.jpg képernyőképen! */}
      <div className="alert alert-dark text-center fw-bold shadow-sm mb-4">
        Vizsgázó neve: asd
      </div>

      <h2 className="mb-4 text-center fw-bold text-success">Add New Darts Player</h2>
      
      <form onSubmit={handleSubmit} className="card p-4 shadow bg-light">
        {/* ID input mező */}
        <div className="mb-3">
          <label className="form-label fw-bold">Azonosító (ID)</label>
          <input type="number" className="form-control" value={id} onChange={e => setId(e.target.value)} required placeholder="Pl.: 8" />
        </div>

        {/* Név input mező */}
        <div className="mb-3">
          <label className="form-label fw-bold">Játékos Neve</label>
          <input type="text" className="form-control" value={nev} onChange={e => setNev(e.target.value)} required placeholder="Pl.: Dennis Priestley" />
        </div>

        {/* Születési dátum input mező */}
        <div className="mb-3">
          <label className="form-label fw-bold">Születési idő (ÉÉÉÉ-HH-NN)</label>
          <input type="text" className="form-control" value={birthDate} onChange={e => setBirthDate(e.target.value)} required placeholder="Pl.: 1950-07-16" />
        </div>

        {/* VB címek száma input mező */}
        <div className="mb-3">
          <label className="form-label fw-bold">Nyert Világbajnokságok száma</label>
          <input type="number" className="form-control" value={worldChWon} onChange={e => setWorldChWon(e.target.value)} required placeholder="Pl.: 1" />
        </div>

        {/* Profil URL input mező */}
        <div className="mb-3">
          <label className="form-label fw-bold">Profil Link (URL)</label>
          <input type="url" className="form-control" value={profileUrl} onChange={e => setProfileUrl(e.target.value)} required placeholder="Pl.: https://hu.wikipedia.org/..." />
        </div>

        {/* Kép URL input mező */}
        <div className="mb-3">
          <label className="form-label fw-bold">Kép Link (URL)</label>
          <input type="url" className="form-control" value={imageUrl} onChange={e => setImageUrl(e.target.value)} required placeholder="Pl.: https://www.sulla.hu/..." />
        </div>
        
        {/* Űrlap elküldése gomb */}
        <button type="submit" className="btn btn-success w-100 py-2 fw-bold mt-3">
          <i className="bi bi-cloud-arrow-up-fill me-2 fs-5"></i> Játékos mentése és küldése
        </button>
      </form>
    </div>
  );
}
